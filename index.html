document.getElementById('loginBtn').addEventListener('click', function () { const user = document.getElementById('username').value.trim(); const pass = document.getElementById('password').value.trim(); // Master login (Super Distributor) if (user === 'CRTECH' && pass === 'CRTECH') { showServices('Super Distributor'); return; } // Check localStorage for Distributor / Retailer const users = JSON.parse(localStorage.getItem('userAccounts') || '[]'); const match = users.find(u => u.username === user && u.password === pass); if (match) { showServices(match.role); } else { alert('Invalid credentials. Please try again.'); } }); // Function to show services based on role function showServices(role) { document.querySelector('.login').style.display = 'none'; document.getElementById('services').style.display = 'grid'; window.scrollTo({ top: document.getElementById('services').offsetTop, behavior: 'smooth' }); alert('Welcome, ' + role + '!'); }
